```typescript
  const BOM = '\uFEFF';
  const header = 'Data,DescriĂ§ĂŁo,Categoria,Ficha IRPF,Valor (R$)\n';
  
  const rows = [

function formatRow(t: Transaction, category: IRPFCategory): string {
  const date = new Date(t.date).toLocaleDateString('pt-BR');
  const description = `"${t.description.replace(/"/g, '""')}"`;
  const categoryLabel = IRPF_FIELD_LABELS[category];
  const ficha = IRPF_FICHA[category];
  const value = t.amount.toFixed(2).replace('.', ',');
  return `${date},${description},${categoryLabel},${ficha},${value}`;
}
```

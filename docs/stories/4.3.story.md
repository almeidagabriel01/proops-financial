### Analytics вЂ” inserГ§ГЈo server-side

```typescript
// src/lib/analytics/events.ts
export async function trackEvent(
  supabaseAdmin: SupabaseClient,
  userId: string,
  eventName: string,
  properties: Record<string, unknown> = {},
) {
  // fire-and-forget вЂ” nunca bloquear a resposta principal
  supabaseAdmin

    .catch(() => {});
}
```

### Migration analytics

```sql
-- supabase/migrations/009_analytics.sql
CREATE TABLE IF NOT EXISTS public.analytics_events (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id     uuid REFERENCES public.profiles(id) ON DELETE CASCADE,
  event_name  text NOT NULL,
  properties  jsonb DEFAULT '{}',
  created_at  timestamptz DEFAULT now()

-- UsuГЎrios nГЈo leem seus eventos (write-only via service role)

-- Index para queries de aggregate

CREATE INDEX analytics_events_event_name_idx ON public.analytics_events(event_name);
```
